# Util 模块的 CMake 配置

# 查找源文件
file(GLOB_RECURSE UTIL_SOURCES_CPP ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
# 添加静态库
add_library(CustomCard STATIC ${UTIL_SOURCES_CPP})

#公开目录供外界访问
target_include_directories(CustomCard PUBLIC
	${CMAKE_CURRENT_SOURCE_DIR}
)
# 链接库
target_link_libraries(CustomCard PRIVATE
	Qt6::Widgets Qt6::Core
	Client-DataCenter Client-NetWork
)
